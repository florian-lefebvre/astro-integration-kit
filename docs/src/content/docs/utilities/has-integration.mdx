import { Tabs, TabItem } from '@astrojs/starlight/components';

`hasIntegration` checks whether an integration has already been added to the Astro config. For example:

<Tabs>
    <TabItem label="Extended hooks">
	```ts title="my-integration/index.ts" "hasIntegration"
	import { defineIntegration } from "astro-integration-kit";

	export default defineIntegration({
		name: "my-integration",
		setup() {
			return {
				"astro:config:setup": ({ hasIntegration }) => {
					if (hasIntegration("@astrojs/tailwind")) {
						console.log("Tailwind is installed!");
					}
				}
			}
		}
	})
	```
	</TabItem>
	<TabItem label="Standalone utilities">
	```ts title="integration/index.ts" "hasIntegration"
	import type { AstroIntegration } from "astro";
	import { hasIntegration } from "astro-integration-kit";

	export default function myIntegration(): AstroIntegration {
		return {
			name: "my-integration",
			hooks: {
				"astro:config:setup": ({ config }) => {
					if (hasIntegration({
						name: "@astrojs/tailwind",
						config,
					})) {
						console.log("Tailwind is installed!")
					}
				},
			}
		}
	}
	``` 
	</TabItem>
</Tabs>

### Relative position check

The `hasIntegration` utility checks whether an integration was installed, but sometimes two integrations must be installed in an specific order to work together correctly.

For that use-case, there are two variants of this utility that checks an integration relative to the current integration. For example:

<Tabs>
    <TabItem label="Extended hooks">
	```ts title="my-integration/index.ts" "hasPreviousIntegration" "hasFollowingIntegration"
	import { defineIntegration } from "astro-integration-kit";

	export default defineIntegration({
		name: "my-integration",
		setup() {
			return {
				"astro:config:setup": ({ hasPreviousIntegration, hasFollowingIntegration }) => {
					if (hasPreviousIntegration("@astrojs/tailwind")) {
						console.warn("my-integration should be installed before tailwind to work correctly");
					}
					if (hasFollowingIntegration("astro-env")) {
						console.warn("my-integration should be installed after astro-env to work correctly");
					}
				}
			}
		}
	})
	```
	</TabItem>
	<TabItem label="Standalone utilities">
	```ts title="integration/index.ts" "hasPreviousIntegration" "hasFollowingIntegration"
	import type { AstroIntegration } from "astro";
	import { hasPreviousIntegration, hasFollowingIntegration } from "astro-integration-kit";

	export default function myIntegration(): AstroIntegration {
		return {
			name: "my-integration",
			hooks: {
				"astro:config:setup": ({ config }) => {
					if (hasPreviousIntegration({
						current: "my-integration",
						target: "@astrojs/tailwind",
						config,
					})) {
						console.warn("my-integration should be installed before tailwind to work correctly");
					}
					if (hasFollowingIntegration({
						current: "my-integration",
						target: "astro-env",
						config,
					})) {
						console.warn("my-integration should be installed after astro-env to work correctly");
					}
				},
			}
		}
	}
	``` 
	</TabItem>
</Tabs>
